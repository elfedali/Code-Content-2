% Professional LaTeX preamble for Code-Content books
% Compatible with tectonic (auto-downloads from CTAN on first run)

% ------- Page geometry -------------------------------------------------------
\usepackage[a4paper, margin=2.4cm, top=2.8cm, bottom=2.8cm]{geometry}

% ------- Fonts (T1 + lmodern, works without system fonts) -------------------
\usepackage[T1]{fontenc}
\usepackage{lmodern}
\usepackage{microtype}

% ------- Colors -------------------------------------------------------------
\usepackage{xcolor}
\definecolor{AccentBlue}{HTML}{4A6CF7}
\definecolor{AccentPurple}{HTML}{6C63FF}
\definecolor{CodeBg}{HTML}{1E1E2E}
\definecolor{RuleColor}{HTML}{D0D4E8}
\definecolor{MutedGray}{HTML}{666680}
\definecolor{QuoteBg}{HTML}{F0F3FF}

% ------- Hyperlinks ---------------------------------------------------------
\usepackage[
  colorlinks=true,
  linkcolor=AccentPurple,
  urlcolor=AccentBlue,
  citecolor=AccentBlue
]{hyperref}

% ------- Section headings ---------------------------------------------------
\usepackage{titlesec}
\titleformat{\section}
  {\color{AccentBlue}\Large\bfseries}
  {\thesection}{1em}{}
  [\vspace{2pt}{\color{RuleColor}\hrule height 0.8pt}\vspace{2pt}]
\titleformat{\subsection}
  {\color{AccentPurple}\large\bfseries}
  {\thesubsection}{1em}{}
\titleformat{\subsubsection}
  {\bfseries\normalsize}
  {\thesubsubsection}{1em}{}
\titlespacing*{\section}{0pt}{26pt}{10pt}
\titlespacing*{\subsection}{0pt}{18pt}{6pt}
\titlespacing*{\subsubsection}{0pt}{12pt}{4pt}

% ------- Headers & footers --------------------------------------------------
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\color{MutedGray}\slshape\leftmark}
\fancyhead[R]{\small\color{MutedGray}\slshape\rightmark}
\fancyfoot[C]{\small\color{MutedGray}\thepage}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\headrule}{\hbox to\headwidth{\color{RuleColor}\leaders\hrule height\headrulewidth\hfill}}
\renewcommand{\footrulewidth}{0pt}
\setlength{\headheight}{14pt}

% ------- TOC styling --------------------------------------------------------
\usepackage{tocloft}
\renewcommand{\cftsecleader}{\cftdotfill{\cftdotsep}}
\renewcommand{\cftsecfont}{\bfseries}
\renewcommand{\cftsecpagefont}{\bfseries\color{AccentBlue}}
\setlength{\cftbeforesecskip}{4pt}
\renewcommand{\contentsname}{\color{AccentBlue}Table of Contents}

% ------- Dark code blocks (wraps pandoc's Shaded env) -----------------------
\usepackage{fancyvrb}
\usepackage{mdframed}
\usepackage{framed}
\definecolor{shadecolor}{HTML}{1E1E2E}
\renewenvironment{Shaded}{%
  \begin{mdframed}[%
    backgroundcolor=CodeBg,%
    linewidth=0pt,%
    roundcorner=5pt,%
    innerleftmargin=14pt,%
    innerrightmargin=14pt,%
    innertopmargin=10pt,%
    innerbottommargin=10pt,%
    skipabove=10pt,%
    skipbelow=10pt%
  ]%
}{\end{mdframed}}

% ------- Tables -------------------------------------------------------------
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{array}
\renewcommand{\arraystretch}{1.4}

% ------- Lists --------------------------------------------------------------
\usepackage{enumitem}
\setlist[itemize]{itemsep=3pt,parsep=0pt,leftmargin=18pt}
\setlist[enumerate]{itemsep=3pt,parsep=0pt,leftmargin=20pt}

% ------- Line & paragraph spacing ------------------------------------------
\usepackage{setspace}
\setstretch{1.22}
\setlength{\parskip}{6pt}
\setlength{\parindent}{0pt}

% ------- Images -------------------------------------------------------------
\usepackage{graphicx}
\makeatletter
\def\maxwidth{\ifdim\Gin@nat@width>\linewidth\linewidth\else\Gin@nat@width\fi}
\makeatother
\setkeys{Gin}{width=\maxwidth,keepaspectratio}
